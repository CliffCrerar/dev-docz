# Installing SQL server on ubuntu

https://computingforgeeks.com/how-to-install-ms-sql-on-ubuntu-18-04-lts/

sudo apt-get update

sudo apt-get upgrade

sudo reboot

    sudo wget -qO- https://packages.microsoft.com/keys/microsoft.asc | sudo apt-key add -

    echo "deb [arch=amd64] https://packages.microsoft.com/ubuntu/16.04/prod xenial main" | sudo tee /etc/apt/sources.list.d/mssql.list

    sudo apt-get update
sudo apt install libcurl3
sudo apt-get install -y mssql-server
